import { db } from "../config/database/connection.js";

/** Obtém todos os cursos do banco de dados. */
const getAllCourses = async () => {
  return new Promise((resolve, reject) => {
     db.all("SELECT * FROM cursos", [], (err, rows) => {
       if (err) {
         reject(err);
       } else {
         resolve(rows);
       }
     });
  });
 };
